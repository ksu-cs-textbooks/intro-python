x = int(input("Enter a positive integer: "))
while x <= 0:
    print("Invalid input!")
    x = int(input("Enter a positive integer: "))

y = int(input("Enter a positive integer: "))
while y <= 0:
    print("Invalid input!")
    y = int(input("Enter a positive integer: "))

while y <= x:
    for i in range(x - y):
        print("*", end="")
        y = y + 2
    print("")

print("Complete!")
